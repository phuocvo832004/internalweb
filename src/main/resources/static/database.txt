CREATE TABLE Users (
    UserID NUMBER PRIMARY KEY,
    Username VARCHAR2(50),
    Password VARCHAR2(50),
    Full_Name VARCHAR2(100),
    Phone VARCHAR2(20),
    Email VARCHAR2(100),
    Position VARCHAR2(100),
    Department VARCHAR2(100),
    Role VARCHAR2(50)
);

CREATE TABLE Internal_News (
    NewsID NUMBER PRIMARY KEY,
    Title VARCHAR2(200),
    Content CLOB,
    Author VARCHAR2(100),
    Date_Posted TIMESTAMP
);

CREATE TABLE Requests (
    RequestID NUMBER PRIMARY KEY,
    UserID NUMBER REFERENCES Users(UserID),
    Title VARCHAR2(200),
    Request_Type VARCHAR2(100),
    Status VARCHAR2(50),
    Date_Submitted DATE,
    Processed_By VARCHAR2(100),
    Comments VARCHAR2(4000)
);

CREATE TABLE Request_Transfers (
    TransferID NUMBER PRIMARY KEY,
    RequestID NUMBER REFERENCES Requests(RequestID),
    Transferred_By VARCHAR2(100),
    Received_By VARCHAR2(100),
    Transfer_Date DATE,
    Notes VARCHAR2(4000)
);

CREATE TABLE Result (
    ResultID NUMBER PRIMARY KEY,
    RequestID NUMBER REFERENCES Requests(RequestID),
    Sender VARCHAR2(100),
    Content CLOB,
    Sent_Date DATE,
    Notes VARCHAR2(4000)
);

-- Sequence for Users
CREATE SEQUENCE seq_users
START WITH 1
INCREMENT BY 1
NOCACHE
NOCYCLE;

-- Sequence for Internal_News
CREATE SEQUENCE seq_internal_news
START WITH 1
INCREMENT BY 1
NOCACHE
NOCYCLE;

-- Sequence for Requests
CREATE SEQUENCE seq_requests
START WITH 1
INCREMENT BY 1
NOCACHE
NOCYCLE;

-- Sequence for Request_Transfers
CREATE SEQUENCE seq_request_transfers
START WITH 1
INCREMENT BY 1
NOCACHE
NOCYCLE;

-- Sequence for Result
CREATE SEQUENCE seq_result
START WITH 1
INCREMENT BY 1
NOCACHE
NOCYCLE;

-- Trigger for Users
CREATE OR REPLACE TRIGGER trg_users
BEFORE INSERT ON Users
FOR EACH ROW
BEGIN
    :NEW.UserID := seq_users.NEXTVAL;
END;
/

-- Trigger for Internal_News
CREATE OR REPLACE TRIGGER trg_internal_news
BEFORE INSERT ON Internal_News
FOR EACH ROW
BEGIN
    :NEW.NewsID := seq_internal_news.NEXTVAL;
END;
/

-- Trigger for Requests
CREATE OR REPLACE TRIGGER trg_requests
BEFORE INSERT ON Requests
FOR EACH ROW
BEGIN
    :NEW.RequestID := seq_requests.NEXTVAL;
END;
/

-- Trigger for Request_Transfers
CREATE OR REPLACE TRIGGER trg_request_transfers
BEFORE INSERT ON Request_Transfers
FOR EACH ROW
BEGIN
    :NEW.TransferID := seq_request_transfers.NEXTVAL;
END;
/

-- Trigger for Result
CREATE OR REPLACE TRIGGER trg_result
BEFORE INSERT ON Result
FOR EACH ROW
BEGIN
    :NEW.ResultID := seq_result.NEXTVAL;
END;
/
